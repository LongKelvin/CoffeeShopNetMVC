CREATE PROCEDURE [dbo].[GetRevenueStatistic]
    @fromDate [nvarchar](max),
    @toDate [nvarchar](max)
AS
BEGIN
    
    select
    o.CreatedDate as Date,
    sum(od.Quantity*od.UnitPrice) as Revenues,
    sum((od.Quantity*od.UnitPrice)-(od.Quantity*p.OriginalPrice)) as Benefit
    from Orders o
    inner join OrderDetails od
    on o.ID = od.OrderId
    inner join Products p
    on od.ProductID  = p.ID
    where o.CreatedDate <= cast(@toDate as date) and o.CreatedDate >= cast(@fromDate as date)
    group by o.CreatedDate
END
GO
